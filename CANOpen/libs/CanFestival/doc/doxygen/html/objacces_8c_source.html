45 UNS8 accessDictionaryError(UNS16 index, UNS8 subIndex,
46  UNS32 sizeDataDict, UNS32 sizeDataGiven, UNS32 code)
47 {
48 #ifdef DEBUG_WAR_CONSOLE_ON
49  MSG_WAR(0x2B09,"Dictionary index : ", index);
50  MSG_WAR(0X2B10," subindex : ", subIndex);
51  switch (code) {
52  case OD_NO_SUCH_OBJECT:
53  MSG_WAR(0x2B11,"Index not found ", index);
54  break;
55  case OD_NO_SUCH_SUBINDEX :
56  MSG_WAR(0x2B12,"SubIndex not found ", subIndex);
57  break;
58  case OD_WRITE_NOT_ALLOWED :
59  MSG_WAR(0x2B13,"Write not allowed, data is read only ", index);
60  break;
61  case OD_LENGTH_DATA_INVALID :
62  MSG_WAR(0x2B14,"Conflict size data. Should be (bytes) : ", sizeDataDict);
63  MSG_WAR(0x2B15,"But you have given the size : ", sizeDataGiven);
64  break;
65  case OD_NOT_MAPPABLE :
66  MSG_WAR(0x2B16,"Not mappable data in a PDO at index : ", index);
67  break;
68  case OD_VALUE_TOO_LOW :
69  MSG_WAR(0x2B17,"Value range error : value too low. SDOabort : ", code);
70  break;
71  case OD_VALUE_TOO_HIGH :
72  MSG_WAR(0x2B18,"Value range error : value too high. SDOabort : ", code);
73  break;
74  default :
75  MSG_WAR(0x2B20, "Unknown error code : ", code);
76  }
77  #endif
78 
79  return 0;
80 }

82 UNS32 _getODentry( CO_Data* d,
83  UNS16 wIndex,
84  UNS8 bSubindex,
85  void * pDestData,
86  UNS32 * pExpectedSize,
87  UNS8 * pDataType,
88  UNS8 checkAccess,
89  UNS8 endianize)
90 { /* DO NOT USE MSG_ERR because the macro may send a PDO -> infinite
91  loop if it fails. */
92  UNS32 errorCode;
93  UNS32 szData;
94  const indextable *ptrTable;
95  ODCallback_t *Callback;
96 
97  ptrTable = (*d->scanIndexOD)(wIndex, &errorCode, &Callback);
98 
99  if (errorCode != OD_SUCCESSFUL)
100  return errorCode;
101  if( ptrTable->bSubCount <= bSubindex ) {
102  /* Subindex not found */
103  accessDictionaryError(wIndex, bSubindex, 0, 0, OD_NO_SUCH_SUBINDEX);
104  return OD_NO_SUCH_SUBINDEX;
105  }
106 
107  if (checkAccess && (ptrTable->pSubindex[bSubindex].bAccessType & WO)) {
108  MSG_WAR(0x2B30, "Access Type : ", ptrTable->pSubindex[bSubindex].bAccessType);
109  accessDictionaryError(wIndex, bSubindex, 0, 0, OD_READ_NOT_ALLOWED);
110  return OD_READ_NOT_ALLOWED;
111  }
112 
113  if (pDestData == 0) {
114  return SDOABT_GENERAL_ERROR;
115  }
116 
117  if (ptrTable->pSubindex[bSubindex].size > *pExpectedSize) {
118  /* Requested variable is too large to fit into a transfer line, inform *
119  * the caller about the real size of the requested variable. */
120  *pExpectedSize = ptrTable->pSubindex[bSubindex].size;
121  return SDOABT_OUT_OF_MEMORY;
122  }
123 
124  *pDataType = ptrTable->pSubindex[bSubindex].bDataType;
125  szData = ptrTable->pSubindex[bSubindex].size;
126 
127 # ifdef CANOPEN_BIG_ENDIAN
128  if(endianize && *pDataType > boolean && !(
129  *pDataType >= visible_string &&
130  *pDataType <= domain)) {
131  /* data must be transmited with low byte first */
132  UNS8 i, j = 0;
133  MSG_WAR(boolean, "data type ", *pDataType);
134  MSG_WAR(visible_string, "data type ", *pDataType);
135  for ( i = szData ; i > 0 ; i--) {
136  MSG_WAR(i," ", j);
137  ((UNS8*)pDestData)[j++] =
138  ((UNS8*)ptrTable->pSubindex[bSubindex].pObject)[i-1];
139  }
140  *pExpectedSize = szData;
141  }
142  else /* no endianisation change */
143 # endif
144 
145  if(*pDataType != visible_string) {
146  memcpy(pDestData, ptrTable->pSubindex[bSubindex].pObject,szData);
147  *pExpectedSize = szData;
148  }else{
149  /* TODO : CONFORM TO DS-301 :
150  * - stop using NULL terminated strings
151  * - store string size in td_subindex
152  * */
153  /* Copy null terminated string to user, and return discovered size */
154  UNS8 *ptr = (UNS8*)ptrTable->pSubindex[bSubindex].pObject;
155  UNS8 *ptr_start = ptr;
156  /* *pExpectedSize IS < szData . if null, use szData */
157  UNS8 *ptr_end = ptr + (*pExpectedSize ? *pExpectedSize : szData) ;
158  UNS8 *ptr_dest = (UNS8*)pDestData;
159  while( *ptr && ptr < ptr_end){
160  *(ptr_dest++) = *(ptr++);
161  }
162 
163  *pExpectedSize = (UNS32) (ptr - ptr_start);
164  /* terminate string if not maximum length */
165  if (*pExpectedSize < szData)
166  *(ptr) = 0;
167  }
168 
169  return OD_SUCCESSFUL;
170 }

193  return OD_NO_SUCH_SUBINDEX;
194  }

196  MSG_WAR(0x2B25, "Access Type : ", ptrTable->pSubindex[bSubindex].bAccessType);

198  return OD_WRITE_NOT_ALLOWED;

200 
201

203  szData = ptrTable->pSubindex[bSubindex].size;
204 
205  if( *pExpectedSize == 0 ||
206  *pExpectedSize == szData ||
207  /* allow to store a shorter string than entry size */
208  (dataType == visible_string && *pExpectedSize < szData))
209  {
210 #ifdef CANOPEN_BIG_ENDIAN
211  /* re-endianize do not occur for bool, strings time and domains */
212  if(endianize && dataType > boolean && !(
213  dataType >= visible_string &&
214  dataType <= domain))
215  {
216  /* we invert the data source directly. This let us do range

221  {
222  UNS8 tmp =((UNS8 *)pSourceData) [(ptrTable->pSubindex[bSubindex].size - 1) - i];
223  ((UNS8 *)pSourceData) [(ptrTable->pSubindex[bSubindex].size - 1) - i] = ((UNS8 *)pSourceData)[i];
224  ((UNS8 *)pSourceData)[i] = tmp;
225  }
226  }
227 #endif
228  errorCode = (*d->valueRangeTest)(dataType, pSourceData);
229  if (errorCode) {
230  accessDictionaryError(wIndex, bSubindex, szData, *pExpectedSize, errorCode);
231  return errorCode;
232  }
233  memcpy(ptrTable->pSubindex[bSubindex].pObject,pSourceData, *pExpectedSize);
234  /* TODO : CONFORM TO DS-301 :
235  * - stop using NULL terminated strings
236  * - store string size in td_subindex
237  * */
238  /* terminate visible_string with '\0' */
239  if(dataType == visible_string && *pExpectedSize < szData)
240  ((UNS8*)ptrTable->pSubindex[bSubindex].pObject)[*pExpectedSize] = 0;
241 
242  *pExpectedSize = szData;
243 
244  /* Callbacks */
245  if(Callback && Callback[bSubindex]){
246  errorCode = (Callback[bSubindex])(d, ptrTable, bSubindex);
247  if(errorCode != OD_SUCCESSFUL)
248  {
249  return errorCode;
250  }
251  }
252 
253  /* TODO : Store dans NVRAM */
254  if (ptrTable->pSubindex[bSubindex].bAccessType & TO_BE_SAVE){
255  (*d->storeODSubIndex)(d, wIndex, bSubindex);
256  }
257  return OD_SUCCESSFUL;
258  }else{
259  *pExpectedSize = szData;
260  accessDictionaryError(wIndex, bSubindex, szData, *pExpectedSize, OD_LENGTH_DATA_INVALID);
261  return OD_LENGTH_DATA_INVALID;
262  }
263 }
