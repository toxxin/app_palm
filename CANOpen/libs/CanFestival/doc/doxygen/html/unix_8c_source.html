159 CAN_PORT canOpen(s_BOARD *board, CO_Data * d)
160 {
161  int i;
162  for(i=0; i < MAX_NB_CAN_PORTS; i++)
163  {
164  if(!canports[i].used)
165  break;
166  }
167 
168 #ifndef NOT_USE_DYNAMIC_LOADING
169  if (&DLL_CALL(canOpen)==NULL) {
170  fprintf(stderr,"CanOpen : Can Driver dll not loaded\n");
171  return NULL;
172  }
173 #endif
174  CAN_HANDLE fd0 = DLL_CALL(canOpen)(board);
175  if(fd0){
176  canports[i].used = 1;
177  canports[i].fd = fd0;
178  canports[i].d = d;
179  d->canHandle = (CAN_PORT)&canports[i];
180  CreateReceiveTask(&(canports[i]), &canports[i].receiveTask, &canReceiveLoop);
181  return (CAN_PORT)&canports[i];
182  }else{
183  MSG("CanOpen : Cannot open board {busname='%s',baudrate='%s'}\n",board->busname, board->baudrate);
184  return NULL;
185  }
186 }
